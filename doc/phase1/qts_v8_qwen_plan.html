QTS V8 系统是什么
QTS V8 是针对 OKX ETH-USDT 永续合约（5分钟周期）的量化交易系统。Rust + Python 混合双核架构：

• Rust 侧（DeepSeek 负责）：WebSocket 数据接收、共享内存缓冲、特征计算、OKX 签名下单、订单状态机、MCTS 树搜索核心。所有延迟 <1μs 的操作。
• Python 侧（你负责）：AI 模型推断（ResNet/AlphaCast via Triton）、量化策略（OBI_v2/资金费率套利）、门控风控（G1–G4）、Harness 编排、标签积累、监控告警、在线微调。

你通过 import v8_core_engine as vce 调用 DeepSeek 实现的 Rust 模块。数据从 Rust SHM（共享内存） 零拷贝传入你的 PyTorch 模型，延迟 <1μs。你的信号通过 Redis 温路径 传递给 Rust 执行层。

# data/shm_bridge.py
import v8_core_engine as vce
import torch, numpy as np

class ShmBridge:
    def __init__(self, mock_mode: bool = False):
        self.mock_mode = mock_mode
        if not mock_mode:
            self._bridge = vce.ShmBridge()  # 连接 /dev/shm/qts_eth5m

    def get_feature_tensor(self, window_secs: int = 60) -> torch.Tensor:
        if self.mock_mode:
            return torch.randn(12, 50)  # 等待 DeepSeek 期间的占位
        dlpack = self._bridge.get_window(secs=window_secs)
        return torch.from_dlpack(dlpack)  # 零拷贝，不发生内存复制

    def get_numpy_view(self, window_secs: int = 30) -> np.ndarray:
        if self.mock_mode:
            return np.zeros((6, 50), dtype=np.float32)
        ptr, shape = self._bridge.get_raw_ptr(secs=window_secs)
        return np.frombuffer(ptr, dtype=np.float32).reshape(shape)
✓ 验收标准
真实 SHM 读取延迟 <1μs (timeit 100次均值)；get_feature_tensor 返回 shape [T, 50] float32；mock_mode=True 时 Alpha 引擎可正常运行（保证并行开发不阻塞）
⚠️ 不要做
不要在 shm_bridge.py 中直接 open('/dev/shm/qts_eth5m')——必须通过 vce.ShmBridge() 接口，否则会绕过 Rust 的 RwLock 保护，导致内存竞争

# models/resnet/resnet_encoder.py
import torch, torch.nn as nn

class ResidualBlock1D(nn.Module):
    def __init__(self, channels: int, kernel_size: int = 3):
        super().__init__()
        self.conv1 = nn.Conv1d(channels, channels, kernel_size, padding='same')
        self.bn1 = nn.BatchNorm1d(channels)
        self.conv2 = nn.Conv1d(channels, channels, kernel_size, padding='same')
        self.bn2 = nn.BatchNorm1d(channels)
        self.act = nn.LeakyReLU(0.1)

    def forward(self, x):
        residual = x
        return self.act(self.bn2(self.conv2(self.act(self.bn1(self.conv1(x))))) + residual)

class ResNetEncoder(nn.Module):
    """输入: [B, 50, T]  输出: [B, 128]"""
    def __init__(self):
        super().__init__()
        self.stem = nn.Conv1d(50, 128, kernel_size=1)
        self.blocks = nn.Sequential(*[ResidualBlock1D(128) for _ in range(6)])
        self.pool = nn.AdaptiveAvgPool1d(1)  # 多尺度感受野

    def forward(self, x):  # x: [B, 50, T]
        return self.pool(self.blocks(self.stem(x))).squeeze(-1)  # [B, 128]

# 导出 ONNX（放入 Triton Model Repository）
def export_onnx(model, path="triton_model_repository/resnet/1/model.onnx"):
    dummy = torch.zeros(1, 50, 60)
    torch.onnx.export(model, dummy, path,
        input_names=["input_features"], output_names=["embedding_128d"],
        dynamic_axes={"input_features": {0: "batch", 2: "time"}})

# models/alphacast/serve_alphacast.py — 统一 Triton gRPC 客户端
import tritonclient.grpc as triton_grpc
import numpy as np

class AlphaCastClient:
    def __init__(self, url: str = "localhost:8001"):
        self.client = triton_grpc.InferenceServerClient(url=url)

    async def predict(self, features_178d: np.ndarray) -> dict:
        # features_178d: shape [1, 178]
        inp = triton_grpc.InferInput("INPUT_0", features_178d.shape, "FP32")
        inp.set_data_from_numpy(features_178d)
        resp = self.client.infer(model_name="alphacast", inputs=[inp])
        return {
            "pred_return": float(resp.as_numpy("pred_return")[0]),
            "sigma":       float(resp.as_numpy("sigma")[0]),
            "confidence":  float(resp.as_numpy("confidence")[0])
        }

class AlphaCastModel(nn.Module):
    """输入: [B, 60, 178]  输出: pred_return, sigma, confidence"""

asyncio/Tokio 调用规则（必须按此实现）
不能在 Python async 函数中直接 await DeepSeek 的 MctsPool.run_sync()。必须通过 asyncio.get_event_loop().run_in_executor(None, pool.run_sync, state_bytes, rollout_fn) 调用。DeepSeek 的 Tokio runtime 在独立 OS 线程中运行，这是唯一安全的调用方式。
# models/mcts/mcts_worker.py
import asyncio, v8_core_engine as vce
from models.alphacast.serve_alphacast import AlphaCastClient

class MctsWorker:
    def __init__(self):
        self.pool = vce.MctsPool(workers=8, timeout_ms=100)
        self.ac_client = AlphaCastClient()

    def _rollout_fn(self, state_bytes: bytes) -> bytes:
        # 同步函数，在 Tokio worker 线程内被 Rust 回调
        # 必须是同步的，不能是 async def
        features = deserialize_state(state_bytes)
        # 注意：此处不能用 asyncio，直接同步调用 Triton
        result = self.ac_client.predict_sync(features)  # 同步版本
        return serialize_alphacast_output(result)

    async def run_mcts(self, signal: AlphaSignal) -> dict:
        state_bytes = serialize_signal_state(signal)
        loop = asyncio.get_event_loop()
        # 通过 executor 隔离 Tokio runtime 和 asyncio
        result_bytes = await loop.run_in_executor(
            None, self.pool.run_sync, state_bytes, self._rollout_fn
        )
        return deserialize_mcts_result(result_bytes)
        # result: {best_action, path_value, sharpe_estimate, simulations_run}

from scipy.optimize import minimize
import numpy as np

class OnlineCalibrator:
    def __init__(self, update_every: int = 50):
        self.T = 1.0      # Temperature Scaling 系数
        self.T_ema = 1.0  # EMA 平滑版本（α=0.3）
        self.buffer = []
        self.n = update_every

    async def on_trade_complete(self, trace_id: str, actual_return: float):
        # 从 TimescaleDB 查询该 trace_id 的 predicted conf
        pred_conf = await self._get_pred_conf(trace_id)
        self.buffer.append((pred_conf, actual_return > 0))  # 二分类标签
        if len(self.buffer) >= self.n:
            await self._update()
            self.buffer.clear()

    async def _update(self):
        confs, outcomes = zip(*self.buffer)
        result = minimize(self._nll, [self.T], method='L-BFGS-B',
                          bounds=[(0.5, 3.0)])  # T 硬限制 [0.5, 3.0]
        new_T = result.x[0]
        # EMA 平滑，防震荡
        self.T_ema = 0.3 * new_T + 0.7 * self.T_ema
        # 连续 3 次反向变化则暂停并告警
        if self._is_oscillating():
            send_telegram_alert("⚠️ Temperature Scaling 震荡，暂停更新")
            return
        self.T = self.T_ema
        # Triton 热切换（修改 output calibration layer 权重）
        await self._push_to_triton(self.T)  # <100ms，无停机

执行规则与约束
规则 1：不要修改 Rust 接口签名
vce.ShmBridge.get_window()、vce.OkxChannel.place_order() 等接口由 DeepSeek 实现。如果你认为接口需要变更，告知 DeepSeek，双方确认后由 DeepSeek 修改 Rust 代码重新编译 .so，你再更新调用代码。不要绕过接口直接操作 /dev/shm 或 Redis 原始数据结构。
规则 2：Protobuf 字段编号永不修改
schemas/*.proto 的字段编号（= 1, = 2...）在 Phase 0 确定后永不修改。字段名和注释可以加，编号不能动。你的 Python 侧序列化和 DeepSeek 的 Rust 侧反序列化依赖字段编号对齐。
规则 3：_rollout_fn 必须是同步函数
传给 MctsPool.run_sync() 的 rollout_fn 必须是普通同步函数（def，不是 async def）。DeepSeek 的 Tokio worker 在 Python::with_gil() 保护下同步调用它，不支持 asyncio。在 _rollout_fn 内调用 Triton 时使用同步版本的 tritonclient 接口。
规则 4：SHM 只读，不写
你的所有 Python 代码对 SHM 只有读权限。写入 SHM 是 DeepSeek 的 feature_engine.rs 的职责。不要尝试通过 numpy 修改 frombuffer 的返回数组（会触发 Rust RwLock 错误）。
规则 5：CFL 标签 100% 记录，不能跳过
无论信号是否通过门控，counterfactual_labeler.py 必须记录每一条。如果 TimescaleDB 写入失败，先写 Redis backup:cfl:{trace_id}，定时重试写库。标签积累速度直接影响 Phase 3 MetaLabeler 的激活时间。
